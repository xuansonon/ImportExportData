JavaScript: Import and Export Data with Base64 Encoding/Decoding

Source Code

Below is the source code that I developed:

var data1 = "0";
var data2 = "0";
var data3 = "0";
var data4 = "0";
var data5 = "0";

function init() {
  document.getElementById("foundInput1").innerHTML = data1;
  document.getElementById("foundInput2").innerHTML = data2;
  document.getElementById("foundInput3").innerHTML = data3;
  document.getElementById("foundInput4").innerHTML = data4;
  document.getElementById("foundInput5").innerHTML = data5;
}

function exportCode() {
  data1 = document.getElementById("input1").value;
  data2 = document.getElementById("input2").value;
  data3 = document.getElementById("input3").value;
  data4 = document.getElementById("input4").value;
  data5 = document.getElementById("input5").value;
  var data = [data1, data2, data3, data4, data5];
  var exportResult = "";
  for(var i = 0; i < 5; i++) {
    exportResult += data[i];
    if(i != 4) {
      exportResult += "||";
    }
  }
  document.getElementById("exportcodearea").value = (btoa(exportResult));
}


function importCode() {
  var importCode = document.getElementById("importcodearea").value;
  importCode = atob(importCode);
  var data = importCode.split("||");
  data1 = data[0];
  data2 = data[1];
  data3 = data[2];
  data4 = data[3];
  data5 = data[4];
  init();
}

function reset() {
  data1 = "0";
  data2 = "0";
  data3 = "0";
  data4 = "0";
  data5 = "0";
  document.getElementById("input1").value = data1;
  document.getElementById("input2").value = data2;
  document.getElementById("input3").value = data3;
  document.getElementById("input4").value = data4;
  document.getElementById("input5").value = data5;
}
